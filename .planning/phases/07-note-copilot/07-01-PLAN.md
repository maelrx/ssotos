**Files created:** `src/schemas/copilot.py`

Create `src/schemas/copilot.py` with all request/response models:

```python
from pydantic import BaseModel, Field
from uuid import UUID
from typing import Literal

# Base
class CopilotAction(BaseModel):
    note_id: UUID
    instruction: str | None = None  # for propose-patch

# Explain / Summarize
class NoteExplanationResponse(BaseModel):
    note_id: UUID
    markdown: str = Field(description="Markdown explanation")
    referenced_headings: list[str] = Field(default_factory=list)

class NoteSummaryResponse(BaseModel):
    note_id: UUID
    markdown: str = Field(description="100-300 word summary")
    key_points: list[str] = Field(default_factory=list)

# Suggest Links
class LinkSuggestion(BaseModel):
    target_note_path: str
    target_title: str
    reason: str

class SuggestLinksResponse(BaseModel):
    note_id: UUID
    suggestions: list[LinkSuggestion]

# Suggest Tags
class TagSuggestion(BaseModel):
    tag: str
    confidence: float = Field(ge=0.0, le=1.0)
    reason: str

class SuggestTagsResponse(BaseModel):
    note_id: UUID
    suggestions: list[TagSuggestion]

# Suggest Structure
class StructureIssue(BaseModel):
    type: Literal["missing_heading", "deeply_nested", "long_paragraph", "logical_gap"]
    location: str = Field(description="Heading path or line range")
    issue: str
    suggestion: str

class SuggestStructureResponse(BaseModel):
    note_id: UUID
    suggestions: list[StructureIssue]

# Propose Patch
class ProposePatchRequest(BaseModel):
    instruction: str = Field(min_length=1, max_length=1000)

class ProposePatchResponse(BaseModel):
    note_id: UUID
    proposal_id: UUID
    diff: str = Field(description="Unified diff")

# Chat
class ChatMessage(BaseModel):
    role: Literal["user", "assistant"]
    content: str

class ChatRequest(BaseModel):
    note_id: UUID
    message: str
    history: list[ChatMessage] = Field(default_factory=list)

class ChatResponse(BaseModel):
    note_id: UUID
    message: ChatMessage
```

### Acceptance Criteria
- [ ] All models use Pydantic v2 (BaseModel, Field)
- [ ] All request/response models covered for 6 actions + chat
- [ ] No placeholder types — all fields typed

```python
        """Generate summary of note."""
        ...

    async def suggest_links(self, note_id: UUID, db) -> SuggestLinksResponse:
        """Find related notes using hybrid search."""
        ...

    async def suggest_tags(self, note_id: UUID, db) -> SuggestTagsResponse:
        """Suggest tags based on content analysis."""
        ...

    async def suggest_structure(self, note_id: UUID, db) -> SuggestStructureResponse:
        """Analyze note structure for improvement opportunities."""
        ...

    async def propose_patch(self, note_id: UUID, instruction: str, db) -> ProposePatchResponse:
        """Generate diff for requested change, create Proposal in Exchange Zone."""
        ...

    async def chat(self, note_id: UUID, message: str, history: list, db) -> ChatResponse:

        ...
```

For each method:
- Build context via `RetrievalService.build_context_pack(note_id, db)` or `hybrid_search`
- Build PydanticAI prompt with system prompt from SOUL.md
- Use `agent.run()` or structured output for generation
- For `propose_patch`: call `ProposalService.create_proposal()` with generated diff

### System Prompt for Copilot Agent

The copilot uses a SEPARATE system prompt from the general agent:

```
You are the Note Copilot for [workspace]. You help users understand, summarize, and improve their notes.

You are stateless — you don't remember previous conversations about this note.
You have access to the note's full content and context.

Rules:
- Be concise and accurate
- Always reference specific sections/headings when explaining
- Never directly edit a note — only suggest or propose
- For patch proposals, output ONLY a valid unified diff (no markdown, no explanation)
```

```python
"""Copilot REST API — per F11-01 through F11-11."""
from fastapi import APIRouter, Depends
from uuid import UUID
from src.db.database import get_db
from src.services.copilot_service import CopilotService
from src.services.retrieval_service import RetrievalService
from src.services.proposal_service import ProposalService
from src.schemas.copilot import (
    ExplainNoteRequest, ExplainNoteResponse,
    SummarizeNoteRequest, SummarizeNoteResponse,
    SuggestLinksResponse,
    SuggestTagsResponse,
    SuggestStructureResponse,
    ProposePatchRequest, ProposePatchResponse,
    ChatRequest, ChatResponse,
)

router = APIRouter(prefix="/copilot", tags=["copilot"])


def get_copilot_service(
    retrieval: RetrievalService = Depends(),
    proposal: ProposalService = Depends(),
    agent = Depends(get_copilot_agent),  # build_agent() result
) -> CopilotService:
    return CopilotService(retrieval, proposal, agent)


@router.post("/explain/{note_id}", response_model=ExplainNoteResponse)
async def explain_note(note_id: UUID, db=Depends(get_db), svc=Depends(get_copilot_service)):
    return await svc.explain(note_id, db)


@router.post("/summarize/{note_id}", response_model=SummarizeNoteResponse)
async def summarize_note(note_id: UUID, db=Depends(get_db), svc=Depends(get_copilot_service)):
    return await svc.summarize(note_id, db)


@router.post("/suggest-links/{note_id}", response_model=SuggestLinksResponse)
async def suggest_links(note_id: UUID, db=Depends(get_db), svc=Depends(get_copilot_service)):
    return await svc.suggest_links(note_id, db)


@router.post("/suggest-tags/{note_id}", response_model=SuggestTagsResponse)
async def suggest_tags(note_id: UUID, db=Depends(get_db), svc=Depends(get_copilot_service)):
    return await svc.suggest_tags(note_id, db)


@router.post("/suggest-structure/{note_id}", response_model=SuggestStructureResponse)
async def suggest_structure(note_id: UUID, db=Depends(get_db), svc=Depends(get_copilot_service)):
    return await svc.suggest_structure(note_id, db)


@router.post("/propose-patch/{note_id}", response_model=ProposePatchResponse)
async def propose_patch(note_id: UUID, body: ProposePatchRequest, db=Depends(get_db), svc=Depends(get_copilot_service)):
    return await svc.propose_patch(note_id, body.instruction, db)


@router.post("/chat/{note_id}", response_model=ChatResponse)
async def chat(note_id: UUID, body: ChatRequest, db=Depends(get_db), svc=Depends(get_copilot_service)):
    return await svc.chat(note_id, body.message, body.history, db)
```
