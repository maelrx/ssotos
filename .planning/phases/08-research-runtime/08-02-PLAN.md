## Task 1: Create Research API Client

**Files:** `frontend/src/api/research.ts`

### Action

Create `frontend/src/api/research.ts`:

```typescript
import { fetchApi } from './client';

export interface ResearchJob {
  id: string;
  query: string;
  state: string;
  progress: number;
  sources_count: number;
  created_at: string;
}

export interface SourceStatus {
  source_id: string;
  url: string | null;
  status: 'pending' | 'crawled' | 'parsed' | 'failed';
  content_hash: string | null;
  error: string | null;
}

export const researchApi = {
  listJobs: () => fetchApi<{ jobs: ResearchJob[] }>('/research/jobs'),

  getJob: (jobId: string) =>
    fetchApi<ResearchJob>(`/research/jobs/${jobId}`),

  getSources: (jobId: string) =>
    fetchApi<{ sources: SourceStatus[] }>(`/research/jobs/${jobId}/sources`),

  createBrief: (data: {
    query: string;
    goal: string;
    questions: string[];
    scope: string;
    depth: string;
    max_sources: number;
  }) => fetchApi<{ job_id: string }>('/research/briefs', {
    method: 'POST',
    body: JSON.stringify(data),
  }),

  cancelJob: (jobId: string) =>
    fetchApi<{ success: boolean }>(`/research/jobs/${jobId}/cancel`, {
      method: 'POST',
    }),

  getBlueprint: (jobId: string) =>
    fetchApi<{ content: string }>(`/research/jobs/${jobId}/blueprint`),

  getSynthesis: (jobId: string) =>
    fetchApi<{ content: string }>(`/research/jobs/${jobId}/synthesis`),
};
```

### Acceptance Criteria
- [ ] Typed API client for all research endpoints
- [ ] All endpoints covered

## Task 2: Create Research Workspace Component

**Files:** `frontend/src/components/research/ResearchWorkspace.tsx`

### Action

Create `frontend/src/components/research/ResearchWorkspace.tsx`:

```typescript
import { useState } from 'react';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { researchApi } from '../../api/research';
import { JobList } from './JobList';
import { SourceList } from './SourceList';
import { SynthesisViewer } from './SynthesisViewer';

export function ResearchWorkspace() {
  const [selectedJobId, setSelectedJobId] = useState<string | null>(null);

  const { data: jobs } = useQuery({
    queryKey: ['research-jobs'],
    queryFn: researchApi.listJobs,
  });

  const selectedJob = jobs?.jobs.find(j => j.id === selectedJobId);

  return (
    <div className="flex h-full">
      {/* Job list sidebar */}
      <div className="w-64 border-r p-4">
        <h2 className="text-lg font-semibold mb-4">Research Jobs</h2>
        <JobList
          jobs={jobs?.jobs || []}
          selectedJobId={selectedJobId}
          onSelect={setSelectedJobId}
        />
      </div>

      {/* Main content */}
      <div className="flex-1 overflow-auto p-6">
        {selectedJob ? (
          <div className="space-y-6">
            <div className="flex items-center justify-between">
              <h1 className="text-2xl font-bold">{selectedJob.query}</h1>
              <div className="flex gap-2">
                <button className="px-3 py-1.5 text-sm rounded border">
                  Retry
                </button>
                <button className="px-3 py-1.5 text-sm rounded border text-destructive">
                  Cancel
                </button>
              </div>
            </div>

            {/* Progress bar */}
            <div className="space-y-1">
              <div className="flex justify-between text-sm">
                <span>{selectedJob.state}</span>
                <span>{selectedJob.progress}%</span>
              </div>
              <div className="h-2 bg-muted rounded">
                <div
                  className="h-full bg-primary rounded"
                  style={{ width: `${selectedJob.progress}%` }}
                />
              </div>
            </div>

            {/* Tabs: Sources | Synthesis | Blueprint */}
            <SourceList jobId={selectedJob.id} />
          </div>
        ) : (
          <div className="flex items-center justify-center h-full text-muted-foreground">
            Select a research job to view details
          </div>
        )}
      </div>
    </div>
  );
}
```

### Acceptance Criteria
- [ ] Job list sidebar with selection
- [ ] Main content with job details
- [ ] Progress bar
- [ ] Action buttons (retry/cancel)
