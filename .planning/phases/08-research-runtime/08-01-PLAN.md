"""Crawl4AI service for web crawling — per F12-03."""
from dataclasses import dataclass
from typing import AsyncIterator
import hashlib
import asyncio

from crawl4ai import AsyncWebCrawler, BrowserConfig, CrawlerRunConfig

@dataclass
class CrawlResult:
    """Result of crawling a single URL."""
    url: str
    markdown: str
    content_hash: str
    metadata: dict
    success: bool
    error: str | None = None

class CrawlService:
    """Wraps AsyncWebCrawler for research pipeline."""

    def __init__(self) -> None:
        self.browser_config = BrowserConfig(headless=True, verbose=False)
        self.crawl_config = CrawlerRunConfig(
            markdown_generator=...,
            max_depth=1,
            page_timeout=30000,
        )

    async def crawl_url(self, url: str) -> CrawlResult:
        """Crawl single URL, return markdown + metadata."""
        async with AsyncWebCrawler(config=self.browser_config) as crawler:
            result = await crawler.crawl(url, config=self.crawl_config)
            if result.success:
                content_hash = hashlib.sha256(result.markdown.encode()).hexdigest()
                return CrawlResult(
                    url=url,
                    markdown=result.markdown,
                    content_hash=content_hash,
                    metadata=result.metadata,
                    success=True,
                )
            else:
                return CrawlResult(
                    url=url,
                    markdown="",
                    content_hash="",
                    metadata={},
                    success=False,
                    error=result.error,
                )

    async def crawl_urls(self, urls: list[str]) -> list[CrawlResult]:
        """Crawl multiple URLs concurrently."""
        tasks = [self.crawl_url(url) for url in urls]
        return await asyncio.gather(*tasks)
```

"""Docling service for document parsing — per F12-04."""
from dataclasses import dataclass
import hashlib

from docling import DocumentConverter

@dataclass
class ParseResult:
    """Result of parsing a document."""
    source_path: str
    markdown: str
    content_hash: str
    pages: int
    tables: int
    metadata: dict
    success: bool
    error: str | None = None

class DoclingService:
    """Wraps DocumentConverter for document parsing."""

    def __init__(self) -> None:
        self.converter = DocumentConverter()

    async def parse_document(self, file_path: str) -> ParseResult:
        """Parse single document, return markdown + structure."""
        result = self.converter.convert(file_path)
        markdown = result.markdown
        content_hash = hashlib.sha256(markdown.encode()).hexdigest()
        return ParseResult(
            source_path=file_path,
            markdown=markdown,
            content_hash=content_hash,
            pages=len(result.pages),
            tables=len(result.tables),
            metadata={},
            success=True,
        )

    async def parse_documents(self, file_paths: list[str]) -> list[ParseResult]:
        """Parse multiple documents."""
        results = []
        for fp in file_paths:
            try:
                results.append(await self.parse_document(fp))
            except Exception as e:
                results.append(ParseResult(
                    source_path=fp,
                    markdown="",
                    content_hash="",
                    pages=0,
                    tables=0,
                    metadata={},
                    success=False,
                    error=str(e),
                ))
        return results
```

from pydantic import BaseModel, Field
from typing import list

class Citation(BaseModel):
    source_id: str
    url: str | None
    excerpt: str

class Finding(BaseModel):
    topic: str
    summary: str
    confidence: float = Field(ge=0.0, le=1.0)
    citations: list[Citation]

class SynthesisResult(BaseModel):
    """Structured synthesis output."""
    summary: str
    key_findings: list[Finding]
    sources: list[str]
    confidence: float
    gaps: list[str]
    research_query: str

class SynthesisService:
    """Generates synthesis from research artifacts using PydanticAI."""

    def __init__(self) -> None:
        pass

    async def synthesize(
        self,
        artifacts: list[dict],
        query: str,
    ) -> SynthesisResult:
        """Analyze artifacts, generate coherent synthesis."""
        # Build context from artifacts
        context = "\n\n".join(
            f"Source: {a['source_id']}\n{a['content']}"
            for a in artifacts
        )

        prompt = f"""Research query: {query}

Sources:
{context}

Generate a synthesis with:
1. Summary (2-3 paragraphs)
2. Key findings (3-5) with source citations
3. Confidence level (0-1)
4. Research gaps

Return structured JSON."""

        # For v1: basic extraction (PydanticAI agent would go here)
        # Return basic structured response
        return SynthesisResult(
            summary=f"Synthesis of {len(artifacts)} sources regarding: {query}",
            key_findings=[],
            sources=[a['source_id'] for a in artifacts],
            confidence=0.7,
            gaps=["Further investigation needed"],
            research_query=query,
        )
