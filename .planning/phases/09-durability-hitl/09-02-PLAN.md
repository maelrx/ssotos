### Context

Create `src/services/approval_service.py`:

```python
"""Approval service for human-in-the-loop job gates."""
from uuid import UUID, uuid4
from datetime import datetime
from typing import Literal
import structlog

from src.db.session import async_session_maker
from src.db.models import Job

logger = structlog.get_logger(__name__)

class ApprovalService:
    """Manages approval lifecycle for jobs."""

    async def request_approval(
        self,
        job_id: UUID,
        approver_type: Literal["user", "role", "policy"] = "user",
        approver_id: str | None = None,
        reason: str | None = None,
    ) -> str:
        """Request approval for a job.

        Returns approval_id.
        """
        async with async_session_maker() as db:
            from src.db.models import Job

            result = await db.execute(
                select(Job).where(Job.id == job_id)
            )
            job = result.scalar_one_or_none()

            if not job:
                raise ValueError(f"Job {job_id} not found")

            approval_id = str(uuid4())
            job.approval_required = True
            job.approval_id = UUID(approval_id)
            job.status = "awaiting_approval"

            await db.commit()
            logger.info("approval_requested", job_id=str(job_id), approval_id=approval_id)
            return approval_id

    async def approve(
        self,
        approval_id: str,
        resolved_by: str | None = None,
        resolution_note: str | None = None,
    ) -> None:
        """Approve a job, resuming its execution."""
        async with async_session_maker() as db:
            result = await db.execute(
                select(Job).where(Job.approval_id == UUID(approval_id))
            )
            job = result.scalar_one_or_none()

            if not job:
                raise ValueError(f"No job found for approval {approval_id}")

            job.status = "running"
            job.approval_required = False
            # approval_id left intact for audit trail

            await db.commit()
            logger.info("approval_approved", job_id=str(job.id), approval_id=approval_id)

    async def reject(
        self,
        approval_id: str,
        resolved_by: str | None = None,
        resolution_note: str | None = None,
    ) -> None:
        """Reject a job, marking it as failed."""
        async with async_session_maker() as db:
            result = await db.execute(
                select(Job).where(Job.approval_id == UUID(approval_id))
            )
            job = result.scalar_one_or_none()

            if not job:
                raise ValueError(f"No job found for approval {approval_id}")

            job.status = "failed"
            job.error_message = f"Rejected: {resolution_note or 'No reason provided'}"
            job.approval_required = False

            await db.commit()
            logger.warning("approval_rejected", job_id=str(job.id), approval_id=approval_id)
```

### Acceptance Criteria
- [ ] `request_approval()` sets job to `awaiting_approval` status
- [ ] `approve()` resumes job to `running`
- [ ] `reject()` marks job as `failed`
- [ ] Audit trail preserved in job.approval_id

```python
async def pause_for_approval(self, db: AsyncSession, job: Job, approval_id: str) -> None:
    """Pause a running job to wait for human approval."""
    job.status = "awaiting_approval"
    job.approval_required = True
    job.approval_id = UUID(approval_id)

    event = JobEvent(
        id=uuid4(),
        job_id=job.id,
        event_type="awaiting_approval",
        message=f"Job paused for approval: {approval_id}",
        extra={"approval_id": approval_id},
        trace_id=job.trace_id,
    )
    db.add(event)
    await db.commit()

    broadcast_job_event({
        "event_type": "job_awaiting_approval",
        "job_id": str(job.id),
        "approval_id": approval_id,
    }, workspace_id=str(job.workspace_id) if job.workspace_id else None)
```

### Acceptance Criteria

## Task 4: Create approval API endpoints

**Files:** `src/api/approvals.py`

### Action

```python
"""Approval REST API — per F13-07."""
from fastapi import APIRouter, HTTPException, Depends
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from uuid import UUID

from src.db.session import get_db
from src.db.models.job import Job
from src.services.approval_service import ApprovalService

router = APIRouter(prefix="/approvals", tags=["approvals"])


@router.get("")
async def list_approvals(
    status: str | None = None,
    db: AsyncSession = Depends(get_db),
):
    """List all jobs awaiting approval."""
    stmt = select(Job).where(
        Job.approval_required == True,
        Job.approval_id.isnot(None),
    )
    if status == "pending":
        stmt = stmt.where(Job.status == "awaiting_approval")

    result = await db.execute(stmt)
    jobs = result.scalars().all()

    return {
        "approvals": [
            {
                "approval_id": str(job.approval_id),
                "job_id": str(job.id),
                "job_type": job.job_type,
                "status": job.status,
                "created_at": job.created_at.isoformat(),
            }
            for job in jobs
        ]
    }


@router.post("/{approval_id}/approve")
async def approve_job(
    approval_id: UUID,
    resolved_by: str | None = None,
    resolution_note: str | None = None,
    db: AsyncSession = Depends(get_db),
):
    """Approve a job, resuming its execution."""
    svc = ApprovalService()
    try:
        await svc.approve(str(approval_id), resolved_by, resolution_note)
    except ValueError as e:
        raise HTTPException(status_code=404, detail=str(e))

    return {"success": True, "message": "Job approved and resumed"}


@router.post("/{approval_id}/reject")
async def reject_job(
    approval_id: UUID,
    resolved_by: str | None = None,
    resolution_note: str | None = None,
    db: AsyncSession = Depends(get_db),
):
    """Reject a job, marking it as failed."""
    svc = ApprovalService()
    try:
        await svc.reject(str(approval_id), resolved_by, resolution_note)
    except ValueError as e:
        raise HTTPException(status_code=404, detail=str(e))

    return {"success": True, "message": "Job rejected"}
```

Register in `src/app.py` under `/api` prefix.

### Acceptance Criteria
- [ ] `GET /approvals` lists all pending approvals
- [ ] `POST /approvals/{id}/approve` resumes job
- [ ] `POST /approvals/{id}/reject` fails job
